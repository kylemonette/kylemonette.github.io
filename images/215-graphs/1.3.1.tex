\documentclass{standalone}

\usepackage{tikz,pgfplots}

\pgfplotsset{compat=1.18}
\pgfplotsset{
	soldot/.style={
		color=black,only marks,mark=*},
	holdot/.style={
		color=black,fill=white,only marks,mark=*},
	compat=1.12,
	standard/.style={
     		every axis x label/.style={at={(current axis.right of origin)},anchor=north west},
    		every axis y label/.style={at={(current axis.above origin)},anchor=north east}}
}

\usetikzlibrary{calc}
\usetikzlibrary{intersections}
\usetikzlibrary{decorations.markings}
\usetikzlibrary{arrows.meta}
\tikzset{>={Latex[scale=1.2]}}

\begin{document}

\begin{tikzpicture}[scale=0.65]
\begin{axis}[grid=none,
axis lines=middle,
xmin=-4,xmax=4,
ymin=-3,ymax=3,
xtick=\empty,
ytick=\empty,
minor tick={},
xlabel=\(x_1\), ylabel=\(x_2\),
samples=250]
\draw[->,thick] (0,0) -- (2,2) node[above right] {$(2,2)$};
\addplot[soldot] coordinates {(2.1,2.1)};
\draw[->,thick] (0,0) -- (3,-1) node[below left] {$(3,-1)$};
\addplot[soldot] coordinates {(3.1,-1.1)};
\draw[->,thick] (0,0) -- (-2,-1) node[below left] {$(-2,-1)$};
\addplot[soldot] coordinates {(-2.1,-1.1)};
\end{axis}
\end{tikzpicture}

\end{document}
